#BOX AND WHISKER CHART 
sales_technology = np.random.uniform(1000,10000,50)
sales_office_supplies = np.random.uniform(2319,10000,50)
sales_furniture = np.random.uniform(1500,10000,50)

plt.boxplot([sales_technology ,sales_office_supplies ,sales_furniture])

plt.title("Sales for diff product categories", fontdict = {'fontsize':20 , 'fontweight':5, 'color':'Green'})
#labeling the axises 
plt.xlabel("Types of products" , fontdict = {'fontsize':13 , 'fontweight':3, 'color':'Blue'})
plt.ylabel("Sales" ,fontdict = {'fontsize':13 , 'fontweight':4, 'color':'Red'})
plt.xticks((1,2,3),["Technology","Office Supplies","Furniture"])
plt.show()
#the orange line is the median for diff values 
#edge below is first quartile i.e. the 25th percentage value 
#edge above is third quartile or 75 percentile 
#vertical line is called whisker and the upper extreme is the higest value in the dataset and the lower end horizontal line is the lowest value in the data set 
#the little corcles beyond the end of teh whiskers are outliers i.e. they do not line in the 3*IQR range and have significantly diff values
#median divides data in 2 equal halves , becomes the 50th percentile
